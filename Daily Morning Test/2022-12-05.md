## L1 - D10.
- [1차] 비밀지도
```python
def solution(n, arr1, arr2):
    answer = []
    bi_arr1 = []
    bi_arr2 = []
    
    # 2진수 변환, 빈자리 0 채우기
    for num1 in arr1:
        bi_num1 = ''
        while num1 > 0:
            bi_num1 += str(num1 % 2)
            num1 = num1 // 2
        bi_num1 = bi_num1[::-1]
        if len(bi_num1) < n:
            bi_num1 = bi_num1.rjust(n,'0')
        bi_arr1.append(bi_num1)
    
    for num2 in arr2:
        bi_num2 = ''
        while num2 > 0:
            bi_num2 += str(num2 % 2)
            num2 = num2 // 2
        bi_num2 = bi_num2[::-1]
        if len(bi_num2) < n:
            bi_num2 = bi_num2.rjust(n,'0')
        bi_arr2.append(bi_num2) 
        
    # arr1과 arr2 비교하기 (모두 공백만 0)
    for n1 in range(n):
        bi_num = ''
        for n2 in range(n):
            
            if bi_arr1[n1][n2] == bi_arr2[n1][n2] == '0':
                bi_num += ' '
            else:
                bi_num += '#'
        answer.append(bi_num)
```
