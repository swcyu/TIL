def solution(board, moves):
    answer = 0
    
    # 보드 돌리기
    pivot_board = list()
    for i in range(len(board)-1,-1,-1):
        board_i = list()
        for j in range(len(board)-1,-1,-1):
            if board[j][i] != 0:
                board_i.append(board[j][i])
        pivot_board.append(board_i)
    pivot_board = pivot_board[::-1]
    
    # 
    stack = list()
    for move in moves:
        move = move -1
        if pivot_board[move] != []:
            stack.append(pivot_board[move].pop())
        if len(stack) >= 2:
            if stack[-1] == stack[-2]:
                stack.pop()
                stack.pop()
                answer += 2  
    return answer

def solution(board, moves):
    answer = 0
    basket = []
    for move in moves:
        # 인형 집기
        for i in range(len(board)):
            doll = board[i][move-1]            
            if doll > 0:
                board[i][move-1] = 0
                break
        # 바구니에 넣기
        if len(basket) > 0 and doll == basket[len(basket)-1]:
            basket.pop()
            answer += 2
        elif doll != 0:
            basket.append(doll)
    return answer
