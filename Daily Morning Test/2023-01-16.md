def solution(n):
    answer = 0
    for num in range(1, n+1):
        check = num
        number = num
        while check < n:
            check += (number + 1)
            number += 1
        if check == n:

def solution(num:int) -> int:
    count = 0
    for n in range(1, num+1):
        div, mod = divmod(num, n)
        if mod == 0 and div % 2 == 1:
            count += 1
    return count

def solution(n):
    answer = 1
    for i in range(1, n+1):
        add = 0
        for j in range(i, n+1):
            if add < n:
                add += j
            elif add == n:
                answer += 1
                break
            else:
                add = 0
                break
    return answer
